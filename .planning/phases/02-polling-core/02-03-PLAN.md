2. Track known message IDs:
   ```javascript
   // Build set of message IDs on page load
   let knownMessageIds = new Set();
   document.querySelectorAll('[data-message-id]').forEach(el => {
       knownMessageIds.add(parseInt(el.dataset.messageId));
   });
   ```

3. Modify `addMessageToUI(message, senderType)` to also:
   - Accept message object with id property
   - Set data-message-id on the created div
   - Add message.id to knownMessageIds
   - Example: `messageDiv.dataset.messageId = message.id;`

4. Add message update function:
   ```javascript
   function updateMessages(messages) {
       const container = document.getElementById('messagesContainer');
       let hasNewMessages = false;

       messages.forEach(msg => {
           if (!knownMessageIds.has(msg.id)) {
               // New message - append to container
               addMessageToUI(msg, msg.sender_type);
               knownMessageIds.add(msg.id);
               hasNewMessages = true;
           }
       });

       // Scroll to bottom if new messages added
       if (hasNewMessages) {
           scrollToBottom();
       }

       // Remove empty state if messages exist
       const emptyState = container.querySelector('.empty-messages');
       if (emptyState && messages.length > 0) {
           emptyState.remove();
       }
   }
   ```

5. Initialize PollingManager for messages:
   ```javascript
   const messagePoller = new PollingManager({
       fetchFn: async (signal) => {
           const response = await fetch(
               `/chatbot/api/conversations/${conversationId}/messages`,
               { signal }
           );
           if (!response.ok) throw new Error(`HTTP ${response.status}`);
           return response.json();
       },
       onUpdate: (data) => {
           updateMessages(data.messages);
       },
       interval: 10000  // 10 seconds (faster for active conversation)
   });

   // Start polling on page load
   document.addEventListener('DOMContentLoaded', () => {
       messagePoller.start();
   });
   ```

6. Update `addMessageToUI` to properly handle the message object:

   - Or accept full message object with sender_type, content, id, sent_at
   - Detect which format by checking if first arg has .sender_type property

7. When user sends a message (in sendMessage function):
   - After successful POST, the returned message has an id
   - Add that id to knownMessageIds immediately to prevent duplicate on next poll
   - The existing addMessageToUI call already adds to DOM

Key implementation details:

- Only APPEND new messages, never replace existing
- Track IDs to prevent duplicates (important if user sends while poll is in-flight)
- Scroll to bottom only when new messages appear
- Handle the empty-messages div removal
